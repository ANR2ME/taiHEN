4 Quick Start
5 --------------------------------------------------------------------------------
6 Here are a couple of examples that demonstrate how to use the taiHEN APIs.
7 
8 ### Do something on startup
9 
10 Configure the following user plugin to load on an application named "AppName"
11 with the title id "ABCD01234".
12 
13 ```c
14 // handle to our hook
15 static tai_hook_ref_t app_start_ref;
16 // our hook for app entry
17 int hook_app_start(SceSize argc, const void *args) {
18  printf("hello world!\n");
19  return TAI_CONTINUE(int, app_start_ref, argc, args);
20 }
21 // our own plugin entry
22 int module_start(SceSize argc, const void *args) {
23  taiHookFunctionExport(&app_start_ref, // Output a reference
24  "AppName", // Name of module being hooked
25  TAI_ANY_LIBRARY, // If there's multiple libs exporting this
26  0x935CD196, // Special NID specifying module_start
27  hook_app_start); // Name of the hook function
28  return SCE_KERNEL_START_SUCCESS;
29 }
30 ```
31 
32 We can build this as `myplugin.suprx` and add it to `ux0:tai/config.txt` under
33 the section `*ABCD01234` and it will be loaded when `ABCD01234` is started and
34 insert the hook.

36 ### Logging Filesystem
37 
38 The following example will log all file opens from applications. Compile it as
39 `kernellog.skprx` and add to `*KERNEL` section in `ux0:tai/config.txt`.
40 
41 ```c
42 // handle to our hook
43 static tai_hook_ref_t open_ref;
44 // this function is in kernel space
45 SceUID hook_user_open(const char *path, int flags, SceMode mode, void *args) {
46  char k_path[256];
47  SceUID fd;
48  fd = TAI_CONTINUE(SceUID, open_ref, path, flags, mode, args);
49  // we need to copy the user pointer to kernel space
50  sceKernelStrncpyUserToKernel(k_path, (uintptr_t)path, 256);
51  // do some logging
52  printf("opening: %s, res: %x\n", k_path, fd);
53  return fd;
54 }
55 // plugin entry
56 int module_start(SceSize argc, const void *args) {
57  taiHookFunctionExportForKernel(KERNEL_PID, // Kernel process
58  &open_ref, // Output a reference
59  "SceIofilemgr", // Name of module being hooked
60  TAI_ANY_LIBRARY, // If there's multiple libs exporting this
61  0xCC67B6FD, // NID specifying sceIoOpen
62  hook_user_open); // Name of the hook function
63  return SCE_KERNEL_START_SUCCESS;
64 }
65 ```

67 ### Chain of hooks
68 
69 Consider one kernel plugin with the code above. Now consider a second kernel
70 plugin as follows.
71 
72 ```c
73 // handle to our hook
74 static tai_hook_ref_t another_open_ref;
75 // this function is in kernel space
76 SceUID hook_user_open_differently(const char *path, int flags, SceMode mode, void *args) {
77  char k_path[256];
78  SceUID fd;
79  fd = TAI_CONTINUE(SceUID, another_open_ref, path, flags, mode, args);
80  // we need to copy the user pointer to kernel space
81  sceKernelStrncpyUserToKernel(k_path, (uintptr_t)path, 256);
82  // filter out certain paths
83  if (strcmp(k_path, "ux0:hidden_file.bin") == 0 && fd >= 0) {
84  sceIoClose(fd); // close the handle
85  fd = SCE_KERNEL_ERROR_NOENT;
86  }
87  return fd;
88 }
89 // another plugin entry
90 int module_start(SceSize argc, const void *args) {
91  taiHookFunctionExportForKernel(KERNEL_PID, // Kernel process
92  &another_open_ref, // Output a reference
93  "SceIofilemgr", // Name of module being hooked
94  TAI_ANY_LIBRARY, // If there's multiple libs exporting this
95  0xCC67B6FD, // NID specifying sceIoOpen
96  hook_user_open_differently); // Name of the hook function
97  return SCE_KERNEL_START_SUCCESS;
98 }
99 ```
100 
101 Now we have both filesystem filtering _and_ logging.

103 ### Enabling dynarec
104 
105 This plugin will be loaded in kernel and changes the return value of a function
106 that does a check to enable dynarec.
107 
108 ```c
109 // handle to our hook
110 static tai_hook_ref_t some_sysroot_check_hook;
111 // patch function
112 static int some_sysroot_check_patched(void) {
113  // It is important that we always call `TAI_CONTINUE` regardless if we need
114  // the return value or not. This ensures other hooks in the chain can run!
115  TAI_CONTINUE(int, some_sysroot_check_hook);
116  return 1;
117 }
118 // plugin entry
119 int module_start(SceSize argc, const void *args) {
120  taiHookFunctionExportForKernel(KERNEL_PID, // Kernel process
121  &some_sysroot_check_hook, // Output a reference
122  "SceSysmem", // Name of module being hooked
123  0x3691DA45, // NID specifying SceSysrootForKernel
124  0xF8769E86, // NID of the export function we patch
125  some_sysroot_check_patched); // Name of the hook function
126  return SCE_KERNEL_START_SUCCESS;
127 }
128 ```

130 ### Local file logging
131 
132 The above examples are all global hooks: every call regardless of origin will be
133 hooked. You can also insert local hooks: a library import from a module. In this
134 example, we have a user plugin loaded with "AppName" that only logs `sceIoOpen`
135 calls from that application.
136 
137 ```c
138 // handle to our hook
139 static tai_hook_ref_t local_open_hook;
140 // this function is in user space now
141 SceUID hook_local_open(const char *path, int flags, SceMode mode) {
142  SceUID fd;
143  fd = TAI_CONTINUE(SceUID, local_open_hook, path, flags, mode);
144  printf("open in AppName: %s, ret: %x", path, fd);
145  return fd;
146 }
147 // our own plugin entry
148 int module_start(SceSize argc, const void *args) {
149  taiHookFunctionImport(&local_open_hook, // Output a reference
150  "AppName", // Name of module being hooked
151  0xCAE9ACE6, // NID specifying SceLibKernel, a wrapper library
152  0x6C60AC61, // NID specifying sceIoOpen
153  hook_local_open); // Name of the hook function
154  return SCE_KERNEL_START_SUCCESS;
155 }
156 ```

158 Advanced Usage
159 --------------------------------------------------------------------------------
160 Below are some common tasks and paradigms that might be of interest to people
161 writing advanced hooks.
162 
163 ### Hooking shared libraries
164 
165 Currently taiHEN does not support hooking the exports of shared libraries (or
166 the imports from shared libraries). Shared libraries are ones loaded in the
167 `0xE0000000` address range and are mainly in modules that most applications
168 import from (example being `SceLibKernel` and `SceGxm`). You can, however, hook
169 imports of shared libraries as usual. So, if multiple non-shared modules import
170 a function of interest, you must hook all of them (if desired).
171 
172 ### Hooking weak imports
173 
174 Weak imports are imports of modules not loaded at application startup and are
175 loaded by `sceSysmoduleLoadModule` (in `SceSysmodule`) or
176 `sceKernelLoadStartModule` (in `SceLibKernel`) or some other function. They
177 typically are loaded on-demand and unloaded when no longer needed to save
178 memory. That means that your hooks must be created after the module is loaded
179 and removed before the module is unloaded. To do this, you have to first hook
180 the module load and unload imports and in those hooks, you create/remove the
181 desired hook. In this example, we wish to hook `sceScreenShotDisable` which is
182 loaded by the game from `sceSysmoduleLoadModule`.
183 
184 ```c
185 // handle to our hooks
186 static tai_hook_ref_t load_hook;
187 static tai_hook_ref_t unload_hook;
188 static tai_hook_ref_t ss_disable_hook;
189 static SceUID ss_disable_uid;
190 // hook to never disable screenshots
191 int hook_ss_disable(void) {
192  int ret;
193  TAI_CONTINUE(int, ss_disable_hook); // so others get a chance to hook
194  ret = sceScreenShotEnable(); // we always re-enable ss
195  return ret;
196 }
197 // hook load module
198 int hook_sysmodule_load(uint16_t id) {
199  int ret;
200  ret = TAI_CONTINUE(int, load_hook, id);
201  if (ret >= 0) { // load successful
202  switch (id) {
203  case SCE_SYSMODULE_SCREEN_SHOT:
204  ss_disable_uid =
205  taiHookFunctionImport(&ss_disable_hook, // Output a reference
206  "AppName", // Name of module being hooked
207  0xF26FC97D, // NID specifying SceScreenShot
208  0x50AE9FF9, // NID specifying sceScreenShotDisable
209  hook_ss_disable); // Name of the hook function
210  break;
211  // you can consider other loaded modules too here ...
212  default:
213  break;
214  }
215  }
216  return ret;
217 }
218 // hook unload module
219 int hook_sysmodule_unload(uint16_t id) {
220  int ret;
221  ret = TAI_CONTINUE(int, unload_hook, id);
222  if (ret >= 0) { // unload successful
223  switch (id) {
224  case SCE_SYSMODULE_SCREEN_SHOT:
225  if (ss_disable_uid >= 0) {
226  taiHookRelease(ss_disable_uid, ss_disable_hook);
227  ss_disable_uid = -1;
228  }
229  break;
230  // you can consider other loaded modules too here ...
231  default:
232  break;
233  }
234  }
235  return ret;
236 }
237 // our own plugin entry
238 int module_start(SceSize argc, const void *args) {
239  ss_disable_uid = -1;
240  taiHookFunctionImport(&load_hook, // Output a reference
241  "AppName", // Name of module being hooked
242  0x03FCF19D, // NID specifying SceSysmodule
243  0x79A0160A, // NID specifying sceSysmoduleLoadModule
244  hook_sysmodule_load); // Name of the hook function
245  taiHookFunctionImport(&unload_hook, // Output a reference
246  "AppName", // Name of module being hooked
247  0x03FCF19D, // NID specifying SceSysmodule
248  0x31D87805, // NID specifying sceSysmoduleUnloadModule
249  hook_sysmodule_unload); // Name of the hook function
250  return SCE_KERNEL_START_SUCCESS;
251 }
252 ```

254 ### Loading a kernel plugin on demand
255 
256 You can specify `suprx` to load with specific titles, but you cannot do so with
257 `skprx`. This is a limitation of the Vita kernel. A workaround is to use the
258 taiHEN APIs to manually load and unload the kernel module directly. Please note
259 this feature only works with unsafe homebrew enabled. Be aware that
260 `module_stop` of your user plugin is not automatically called so you should not
261 use that to cleanup your kernel plugin. Instead, your kernel plugin's
262 `module_start` can return `SCE_KERNEL_START_NO_RESIDENT` to be cleaned up
263 automatically after running. In the case that it is not possible, you should be
264 careful not to load the same kernel module twice.
265 
266 It it important to remember to always clean up hooks and injections in
267 `module_stop` for kernel modules. You should be doing this for user modules as
268 well, but patches in user-space will be cleaned up by taiHEN when the process
269 exits. Patches in kernel will not be cleaned up automatically.
270 
271 ### Syscall stack limitations
272 
273 When writing a kernel module that exposes new syscalls, know that the syscall
274 stack is only 4096 bytes. That means you might easily run out of space. You
275 should use `sceKernelRunWithStack` to increase the stack size if needed.
276 
277 ### Unloading a kernel module that exposes syscalls
278 
279 By default, the Vita does not allow unloading modules that exposes syscalls (you
280 will get `SCE_KERNEL_ERROR_MODULEMGR_IN_USE`). To get around this, you can hook
281 an override function in the module manager. This hook will be released when the
282 module is unloaded.
283 
284 ```c
285 // handles
286 static hook_ref_t unload_allowed_hook;
287 static SceUID unload_allowed_uid;
288 // patch
289 int unload_allowed_patched(void) {
290  int ret;
291  ret = TAI_CONTINUE(int, unload_allowed_hook);
292  return 1; // always allowed
293 }
294 // kernel plugin entry
295 int module_start(SceSize argc, const void *args) {
296  unload_allowed_uid =
297  taiHookFunctionImportForKernel(KERNEL_PID,
298  &unload_allowed_hook, // Output a reference
299  "SceKernelModulemgr", // Name of module being hooked
300  0x11F9B314, // NID specifying SceSblACMgrForKernel
301  0xBBA13D9C, // Function NID
302  unload_allowed_patched); // Name of the hook function
303  // do other things here...
304  return SCE_KERNEL_START_SUCCESS;
305 }
306 // cleanup
307 int module_stop(SceSize argc, const void *args) {
308  // don't forget to clean up
309  taiHookReleaseForKernel(KERNEL_PID, unload_allowed_uid, unload_allowed_hook);
310  // do other things here...
311  return SCE_KERNEL_STOP_SUCCESS;
312 }
313 ```

315 ### Hooking a function while it is being called
316 
317 There is currently a [bug](https://github.com/yifanlu/taiHEN/issues/12) in the
318 implementation where a crash may happen if you are hooking a function while it
319 is being used. Usually this doesn't happen (except by chance), but some
320 functions might be called in a tight loop. A temporary workaround is listed in
321 the [issue tracker](https://github.com/yifanlu/taiHEN/issues/12).
