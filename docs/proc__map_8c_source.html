109 int proc_map_try_insert(tai_proc_map_t *map, tai_patch_t *patch, tai_patch_t **existing) {
110  int idx;
111  int overlap;
112  tai_proc_t **item, *proc;
113  tai_patch_t **cur, *tmp;
114 
115  idx = patch->pid % map->nbuckets;
116  *existing = NULL;
117 
118  // get proc structure if found
119  sceKernelLockMutexForKernel(map->lock, 1, NULL);
120  item = &map->buckets[idx];
121  while (*item != NULL && (*item)->pid < patch->pid) {
122  item = &(*item)->next;
123  }
124  if (*item != NULL && (*item)->pid == patch->pid) {
125  // existing block
126  proc = *item;
127  } else {
128  // new block
129  proc = sceKernelMemPoolAlloc(g_map_pool, sizeof(tai_proc_t));
130  proc->pid = patch->pid;
131  proc->head = NULL;
132  proc->next = *item;
133  slab_init(&proc->slab, g_exe_slab_item_size, patch->pid);
134  *item = proc;
135  }
136 
137  // now insert into range if needed
138  cur = &proc->head;
139  overlap = 0;
140  while (*cur != NULL) {
141  tmp = *cur;
142  if (tmp->addr < patch->addr) {
143  if (tmp->addr + tmp->size <= patch->addr) {
144  // cur block is completely before our block
145  cur = &(*cur)->next;
146  continue;
147  } else {
148  // cur block's end overlaps with our block's start
149  overlap = 1;
150  }
151  } else if (patch->addr < tmp->addr) {
152  if (patch->addr + patch->size > tmp->addr) {
153  // cur block's start over overlaps with our block's end
154  overlap = 1;
155  } else {
156  // cur block is completely after our block, this is where we insert
157  }
158  } else { // tmp->addr == patch->addr
159  if (patch->size == tmp->size) {
160  // completely overlap
161  overlap = 1;
162  *existing = tmp;
163  }
164  }
165  break;
166  }
167  if (!overlap) {
168  patch->next = *cur;
169  patch->slab = &proc->slab;
170  *cur = patch;
171  }
172  sceKernelUnlockMutexForKernel(map->lock, 1);
173  return !overlap;
174 }

197 int proc_map_remove_all_pid(tai_proc_map_t *map, SceUID pid, tai_patch_t **head) {
198  int idx;
199  tai_proc_t **cur, *tmp;
200 
201  idx = pid % map->nbuckets;
202  *head = NULL;
203  sceKernelLockMutexForKernel(map->lock, 1, NULL);
204  cur = &map->buckets[idx];
205  while (*cur != NULL && (*cur)->pid < pid) {
206  cur = &(*cur)->next;
207  }

209  tmp = *cur;

212  slab_destroy(&tmp->slab);

214  }
215  sceKernelUnlockMutexForKernel(map->lock, 1);
216  return *head != NULL;

228 int proc_map_remove(tai_proc_map_t *map, tai_patch_t *patch) {
229  int idx;
230  int found;
231  tai_proc_t **proc, *next;
232  tai_patch_t **cur;
233 
234  idx = patch->pid % map->nbuckets;
235  found = 0;
236  sceKernelLockMutexForKernel(map->lock, 1, NULL);
237  proc = &map->buckets[idx];
238  while (*proc != NULL && (*proc)->pid < patch->pid) {
239  proc = &(*proc)->next;
240  }
241  if (*proc != NULL && (*proc)->pid == patch->pid) {
242  cur = &(*proc)->head;
243  while (*cur != NULL && *cur != patch) {
244  cur = &(*cur)->next;
245  }
246  if (*cur != NULL) {
247  *cur = patch->next;
248  found = 1;
249  }
250  }
251  if (*proc != NULL && (*proc)->head == NULL) { // it's now empty
252  patch->slab = NULL; // remove reference
253  next = (*proc)->next;
254  slab_destroy(&(*proc)->slab);
255  sceKernelMemPoolFree(g_map_pool, *proc);
256  *proc = next;
257  }
258  sceKernelUnlockMutexForKernel(map->lock, 1);
259  return found;
260 }
