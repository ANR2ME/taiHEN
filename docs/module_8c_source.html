108 static int sce_to_tai_module_info(SceUID pid, void *sceinfo, tai_module_info_t *taiinfo) {
109  SceKernelFwInfo fwinfo;
110  char *info;
111 
112  if (fw_version == 0) {
113  fwinfo.size = sizeof(fwinfo);
114  if (sceKernelGetSystemSwVersion(&fwinfo) < 0) {
115  fw_version = DEFAULT_FW_VERSION;
116  } else {
117  fw_version = fwinfo.version;
118  }
119  LOG("sceKernelGetSystemSwVersion: 0x%08X", fw_version);
120  }
121 
122  if (taiinfo->size < sizeof(tai_module_info_t)) {
123  LOG("Structure size too small: %d", taiinfo->size);
124  return TAI_ERROR_SYSTEM;
125  }
126 
127  info = (char *)sceinfo;
128  if (fw_version >= 0x3600000) {
129  if (pid == KERNEL_PID) {
130  taiinfo->modid = *(SceUID *)(info + 0xC);
131  snprintf(taiinfo->name, 27, "%s", *(const char **)(info + 0x1C));
132  } else {
133  taiinfo->modid = *(SceUID *)(info + 0x10);
134  sceKernelStrncpyUserForPid(pid, taiinfo->name, *(uintptr_t *)(info + 0x1C), 27);
135  }
136  taiinfo->name[26] = '\0';
137  taiinfo->module_nid = *(uint32_t *)(info + 0x30);
138  taiinfo->exports_start = *(uintptr_t *)(info + 0x20);
139  taiinfo->exports_end = *(uintptr_t *)(info + 0x24);
140  taiinfo->imports_start = *(uintptr_t *)(info + 0x28);
141  taiinfo->imports_end = *(uintptr_t *)(info + 0x2C);
142  } else if (fw_version >= 0x1692000) {
143  if (pid == KERNEL_PID) {
144  taiinfo->modid = *(SceUID *)(info + 0x0);
145  } else {
146  taiinfo->modid = *(SceUID *)(info + 0x4);
147  }
148  taiinfo->module_nid = *(uint32_t *)(info + 0x3C);
149  snprintf(taiinfo->name, 27, "%s", (const char *)(info + 0xC));
150  taiinfo->name[26] = '\0';
151  taiinfo->exports_start = *(uintptr_t *)(info + 0x2C);
152  taiinfo->exports_end = *(uintptr_t *)(info + 0x30);
153  taiinfo->imports_start = *(uintptr_t *)(info + 0x34);
154  taiinfo->imports_end = *(uintptr_t *)(info + 0x38);
155  } else {
156  LOG("Unsupported FW 0x%08X", fw_version);
157  return TAI_ERROR_SYSTEM;
158  }
159  return TAI_SUCCESS;
160 }

174 static size_t find_int_for_user(SceUID pid, uintptr_t src, uint32_t needle, size_t size) {
175  int context[3];
176  int flags;
177  uintptr_t end;
178  uint32_t data;
179  size_t count;
180 
181  count = 0;
182  end = (src + size) & ~3; // align to last 4 byte boundary
183  src = (src + 3) & ~3; // align to next 4 byte boundary
184  if (end <= src) {
185  return 0;
186  }
187  size = end-src;
188  flags = sceKernelCpuDisableInterrupts();
189  sceKernelCpuSaveContext(context);
190  while (count < size) {
191  asm ("ldrt %0, [%1]" : "=r" (data) : "r" (src+count));
192  if (data == needle) {
193  break;
194  }
195  count += 4;
196  }
197  sceKernelCpuRestoreContext(context);
198  sceKernelCpuEnableInterrupts(flags);
199  if (size == 0) {
200  return 0;
201  } else {
202  return count;
203  }
204 }

221 int module_get_by_name_nid(SceUID pid, const char *name, uint32_t nid, tai_module_info_t *info) {
222  SceUID modlist[MOD_LIST_SIZE];
223  void *sceinfo;
224  size_t count;
225  int ret;
226 
227  count = MOD_LIST_SIZE;
228  ret = sceKernelGetModuleListForKernel(pid, 1, pid == KERNEL_PID ? 1 : 2, modlist, &count);
229  LOG("sceKernelGetModuleListForKernel(%x): 0x%08X, count: %d", pid, ret, count);
230  if (ret < 0) {
231  return ret;
232  }
233  for (int i = 0; i < count; i++) {
234  ret = sceKernelGetModuleInternal(modlist[i], &sceinfo);
235  //LOG("sceKernelGetModuleInternal(%x): 0x%08X", modlist[i], ret);
236  if (ret < 0) {
237  LOG("Error getting info for mod: %x", modlist[i]);
238  continue;
239  }
240  if (sce_to_tai_module_info(pid, sceinfo, info) < 0) {
241  continue;
242  }
243  if (pid != KERNEL_PID)
244  LOG("Module %s, NID:0x%08X", info->name, info->modid);
245  if (name != NULL && strncmp(name, info->name, 27) == 0) {
246  if (nid == 0 || info->modid == nid) {
247  LOG("Found module %s, NID:0x%08X", name, info->modid);
248  return TAI_SUCCESS;
249  }
250  } else if (name == NULL && info->modid == nid) {
251  LOG("Found module %s, NID:0x%08X", info->name, info->modid);
252  return TAI_SUCCESS;
253  }
254  }
255 
256  return TAI_ERROR_NOT_FOUND;
257 }

270 int module_get_offset(SceUID pid, SceUID modid, int segidx, size_t offset, uintptr_t *addr) {
271  SceKernelModuleInfo sceinfo;
272  size_t count;
273  int ret;
274 
275  if (segidx > 3) {
276  LOG("Invalid segment index: %d", segidx);
277  return TAI_ERROR_INVALID_ARGS;
278  }
279  LOG("Getting offset for pid:%x, modid:%x, segidx:%d, offset:%x", pid, modid, segidx, offset);
280  if (pid != KERNEL_PID) {
281  modid = sceKernelKernelUidForUserUid(pid, modid);
282  LOG("sceKernelKernelUidForUserUid(%x): 0x%08X", pid, modid);
283  if (modid < 0) {
284  LOG("Cannot find kernel object for user object.");
285  return TAI_ERROR_NOT_FOUND;
286  }
287  }
288  sceinfo.size = sizeof(sceinfo);
289  ret = sceKernelGetModuleInfoForKernel(pid, modid, &sceinfo);
290  LOG("sceKernelGetModuleInfoForKernel(%x, %x): 0x%08X", pid, modid, ret);
291  if (ret < 0) {
292  LOG("Error getting segment info for %d", modid);
293  return ret;
294  }
295  if (offset > sceinfo.segments[segidx].memsz) {
296  LOG("Offset %x overflows segment size %x", offset, sceinfo.segments[segidx].memsz);
297  return TAI_ERROR_INVALID_ARGS;
298  }
299  *addr = (uintptr_t)sceinfo.segments[segidx].vaddr + offset;
300  LOG("found address: 0x%08X", *addr);
301 
302  return TAI_SUCCESS;
303 }

316 int module_get_export_func(SceUID pid, const char *modname, uint32_t libnid, uint32_t funcnid, uintptr_t *func) {
317  sce_module_exports_t local;
318  tai_module_info_t info;
319  sce_module_exports_t *export;
320  uintptr_t cur;
321  size_t found;
322  int i;
323 
324  LOG("Getting export for pid:%x, modname:%s, libnid:%d, funcnid:%x", pid, modname, libnid, funcnid);
325  info.size = sizeof(info);
326  if (module_get_by_name_nid(pid, modname, 0, &info) < 0) {
327  LOG("Failed to find module: %s", modname);
328  return TAI_ERROR_NOT_FOUND;
329  }
330 
331  for (cur = info.exports_start; cur < info.exports_end; ) {
332  if (pid == KERNEL_PID) {
333  export = (sce_module_exports_t *)cur;
334  } else {
335  sceKernelMemcpyUserToKernelForPid(pid, &local, cur, sizeof(local));
336  export = &local;
337  }
338 
339  if (libnid == 0 || export->lib_nid == libnid) {
340  if (pid == KERNEL_PID) {
341  for (i = 0; i < export->num_functions; i++) {
342  if (export->nid_table[i] == funcnid) {
343  *func = (uintptr_t)export->entry_table[i];
344  LOG("found kernel address: 0x%08X", *func);
345  return TAI_SUCCESS;
346  }
347  }
348  } else {
349  found = find_int_for_user(pid, (uintptr_t)export->nid_table, funcnid, export->num_functions * 4);
350  if (found) {
351  sceKernelMemcpyUserToKernelForPid(pid, func, (uintptr_t)export->entry_table + found, 4);
352  LOG("found user address: 0x%08X", *func);
353  return TAI_SUCCESS;
354  }
355  }
356  }
357  cur += export->size;
358  }
359 
360  return TAI_ERROR_NOT_FOUND;
361 }
