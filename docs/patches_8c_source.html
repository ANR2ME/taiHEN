125 void cache_flush(SceUID pid, uintptr_t vma, size_t len) {
126  uintptr_t vma_align;
127  int flags;
128  int my_context[3];
129  int ret;
130  int *other_context;
131  int dacr;
132 
133  vma_align = vma & ~0x1F;
134  len = ((vma + len + 0x1F) & ~0x1F) - vma_align;
135  LOG("cache flush: vma %p, vma_align %p, len %x", vma, vma_align, len);
136 
137  if (pid == KERNEL_PID) {
138  sceKernelCpuDcacheFlush((void *)vma_align, len);
139  sceKernelCpuIcacheAndL2Flush((void *)vma_align, len);
140  } else {
141  // TODO: Take care of SHARED_PID
142  flags = sceKernelCpuDisableInterrupts();
143  sceKernelCpuSaveContext(my_context);
144  ret = sceKernelGetPidContext(pid, &other_context);
145  if (ret >= 0) {
146  sceKernelCpuRestoreContext(other_context);
147  asm volatile ("mrc p15, 0, %0, c3, c0, 0" : "=r" (dacr));
148  asm volatile ("mcr p15, 0, %0, c3, c0, 0" :: "r" (0x15450FC3));
149  sceKernelCpuDcacheFlush((void *)vma_align, len);
150  sceKernelCpuIcacheAndL2Flush((void *)vma_align, len);
151  asm volatile ("mcr p15, 0, %0, c3, c0, 0" :: "r" (dacr));
152  }
153  sceKernelCpuRestoreContext(my_context);
154  sceKernelCpuEnableInterrupts(flags);
155  LOG("sceKernelSwitchVmaForPid(%d): 0x%08X\n", pid, ret);
156  }
157  asm volatile ("isb" ::: "memory");
158 }

172 static int tai_hook_function(struct slab_chain *slab, void *target_func, const void *src_func, void **old, void **saved) {
173  struct substitute_function_hook hook;
174  int ret;
175 
176  if (target_func == src_func) {
177  LOG("no hook needed");
178  return TAI_SUCCESS; // no need for hook
179  }
180 
181  hook.function = target_func;
182  hook.replacement = (void *)src_func;
183  hook.old_ptr = old;
184  hook.options = 0;
185  hook.opt = slab;
186  LOG("Calling substitute_hook_functions");
187  ret = substitute_hook_functions(&hook, 1, (struct substitute_function_hook_record **)saved, 0);
188  LOG("Done hooking");
189  if (ret != SUBSTITUTE_OK) {
190  LOG("libsubstitute error: %s", substitute_strerror(ret));
191  return TAI_ERROR_HOOK_ERROR;
192  }
193  return TAI_SUCCESS;
194 }

275 static int hooks_add_hook(tai_hook_list_t *hooks, tai_hook_t *item) {
276  tai_hook_t *head;
277  int ret;
278 
279  LOG("Adding hook %p to chain %p", item, hooks);
280  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
281  if (hooks->head == NULL) { // first hook for this list
282  ret = tai_hook_function(item->patch->slab, hooks->func, item->u.func, &hooks->old, &hooks->saved);
283  if (ret >= 0) {
284  hooks->head = item;
285  item->next = NULL;
286  item->u.next = (uintptr_t)NULL;
287  item->u.old = hooks->old;
288  cache_flush(item->patch->pid, slab_getmirror(item->patch->slab, item), sizeof(tai_hook_t));
289  } else {
290  LOG("Hook failed, do not add to chain");
291  }
292  } else {
293  head = hooks->head;
294  item->next = head->next;
295  item->u.next = head->u.next;
296  item->u.old = hooks->old;
297  head->next = item;
298  head->u.next = slab_getmirror(item->patch->slab, item);
299  LOG("Added hook to existing chain %p", head);
300  // flush cache for head + item, which were modified
301  cache_flush(item->patch->pid, slab_getmirror(item->patch->slab, head), sizeof(tai_hook_t));
302  cache_flush(item->patch->pid, head->u.next, sizeof(tai_hook_t));
303  ret = 1;
304  }
305  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
306 
307  return ret;
308 }

324 static int hooks_remove_hook(tai_hook_list_t *hooks, tai_hook_t *item) {
325  tai_hook_t **cur;
326  uintptr_t tmp, *cur_user;
327  int ret;
328 
329  LOG("Removing hook %p for %p", item, hooks);
330  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
331  if (hooks->head == item) { // first hook for this list
332  // we must remove the patch
333  tai_unhook_function(hooks->saved);
334  hooks->saved = NULL;
335  // set head to the next item
336  hooks->head = item->next;
337  if (hooks->head != NULL) {
338  // add a patch to the new head
339  ret = tai_hook_function(item->patch->slab, hooks->func, hooks->head->u.func, &hooks->old, &hooks->saved);
340  // update the old pointers
341  for (cur = &hooks->head; *cur != NULL; cur = &(*cur)->next) {
342  (*cur)->u.old = hooks->old;
343  }
344  // clear cache of mirror for the last item since it uses the old pointer
345  cache_flush(item->patch->pid, (uintptr_t)cur - offsetof(tai_hook_t, next), sizeof(tai_hook_t));
346  } else {
347  ret = 0;
348  }
349  } else {
350  cur = &hooks->head;
351  cur_user = &tmp;
352  ret = -1;
353  while (1) {
354  if (*cur) {
355  if (*cur == item) {
356  *cur = item->next; // remove from list
357  *cur_user = item->u.next;
358  // clear cache since pointers were changed
359  cache_flush(item->patch->pid, (uintptr_t)cur - offsetof(tai_hook_t, next), sizeof(tai_hook_t));
360  ret = 0;
361  break;
362  } else {
363  cur = &(*cur)->next;
364  cur_user = &(*cur)->u.next;
365  }
366  } else {
367  break;
368  }
369  }
370  }
371  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
372  return ret;
373 }

385 SceUID tai_hook_func_abs(tai_hook_ref_t *p_hook, SceUID pid, void *dest_func, const void *hook_func) {
386  tai_patch_t *patch, *tmp;
387  tai_hook_t *hook;
388  int ret;
389  struct slab_chain *slab;
390  uintptr_t exe_addr;
391 
392  LOG("Hooking %p to %p for pid %x", hook_func, dest_func, pid);
393  if (hook_func >= MEM_SHARED_START) {
394  if (pid == KERNEL_PID) {
395  return TAI_ERROR_INVALID_KERNEL_ADDR; // invalid hook address
396  } else {
397  return TAI_ERROR_NOT_IMPLEMENTED; // TODO: add support for this
398  }
399  }
400 
401  hook = NULL;
402  // TODO: create it for a PID to allow auto cleanup on process exit
403  ret = sceKernelCreateUidObj(&g_taihen_class, "tai_patch_hook", NULL, (SceObjectBase **)&patch);
404  LOG("sceKernelCreateUidObj(tai_patch_hook): 0x%08X, %p", ret, patch);
405  if (ret < 0) {
406  return ret;
407  }
408 
409  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
410  patch->type = HOOKS;
411  patch->uid = ret;
412  patch->pid = pid;
413  patch->addr = FUNC_TO_UINTPTR_T(dest_func);
414  patch->size = FUNC_SAVE_SIZE;
415  patch->next = NULL;
416  patch->data.hooks.func = dest_func;
417  patch->data.hooks.saved = NULL;
418  patch->data.hooks.head = NULL;
419  if (proc_map_try_insert(g_map, patch, &tmp) < 1) {
420  ret = sceKernelDeleteUid(patch->uid);
421  LOG("sceKernelDeleteUid(old): 0x%08X", ret);
422  if (tmp == NULL || tmp->type != HOOKS) {
423  // error
424  LOG("this hook overlaps an existing hook");
425  ret = TAI_ERROR_PATCH_EXISTS;
426  goto err;
427  } else {
428  // we have an existing patch
429  LOG("found existing patch %p, discarding %p", tmp, patch);
430  patch = tmp;
431  }
432  }
433 
434  hook = slab_alloc(patch->slab, &exe_addr);
435  if (hook == NULL) {
436  ret = -1;
437  goto err;
438  }
439  hook->u.func = (void *)hook_func;
440  hook->patch = patch;
441 
442  ret = hooks_add_hook(&patch->data.hooks, hook);
443  if (ret < 0 && patch->data.hooks.head == NULL) {
444  LOG("failed to add hook and patch is now empty, freeing it");
445  slab_free(patch->slab, hook);
446  hook = NULL;
447  proc_map_remove(g_map, patch);
448  sceKernelDeleteUid(patch->uid);
449  patch = NULL;
450  } else if (ret >= 0) {
451  ret = patch->uid;
452  *p_hook = slab_getmirror(patch->slab, hook);
453  }
454 
455 err:
456  // error and we have allocated a hook
457  if (ret < 0 && patch && hook) {
458  slab_free(patch->slab, hook);
459  }
460 
461  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
462 
463  return ret;
464 }

474 int tai_hook_release(SceUID uid, tai_hook_ref_t hook_ref) {
475  tai_hook_t **cur, *hook;
476  tai_patch_t *patch;
477  struct slab_chain *slab;
478  int ret;
479 
480  ret = sceKernelGetObjForUid(uid, &g_taihen_class, (SceObjectBase **)&patch);
481  LOG("sceKernelGetObjForUid(%x): 0x%08X", uid, ret);
482  if (ret < 0) {
483  return ret;
484  }
485  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
486  slab = patch->slab;
487  for (cur = &patch->data.hooks.head; *cur != NULL; cur = &(*cur)->next) {
488  if (slab_getmirror(slab, *cur) == hook_ref) {
489  hook = *cur;
490  LOG("Found hook %p for ref %p", hook, hook_ref);
491  ret = hooks_remove_hook(&patch->data.hooks, hook);
492  *cur = hook->next;
493  LOG("freeing hook");
494  slab_free(slab, hook);
495  if (patch->data.hooks.head == NULL) {
496  LOG("patch is now empty, freeing it");
497  proc_map_remove(g_map, patch);
498  sceKernelDeleteUid(patch->uid);
499  }
500  ret = TAI_SUCCESS;
501  goto end;
502  }
503  }
504  LOG("Cannot find hook for uid %x ref %p", uid, hook_ref);
505  ret = TAI_ERROR_NOT_FOUND;
506 end:
507  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
508 
509  return ret;
510 }

525 SceUID tai_inject_abs(SceUID pid, void *dest, const void *src, size_t size) {
526  tai_patch_t *patch, *tmp;
527  void *saved;
528  int ret;
529 
530  // TODO: Check that dest is not inside our slab structure... that could corrupt kernel code
531 
532  LOG("Injecting %p with %p for size 0x%08X at pid %x", dest, src, size, pid);
533  ret = sceKernelCreateUidObj(&g_taihen_class, "tai_patch_inject", NULL, (SceObjectBase **)&patch);
534  LOG("sceKernelCreateUidObj(tai_patch_inject): 0x%08X, %p", ret, patch);
535  if (ret < 0) {
536  return ret;
537  }
538 
539  saved = sceKernelMemPoolAlloc(g_patch_pool, size);
540  LOG("sceKernelMemPoolAlloc(g_patch_pool, 0x%08X): %p", size, saved);
541  if (saved == NULL) {
542  sceKernelDeleteUid(ret);
543  return TAI_ERROR_MEMORY;
544  }
545 
546  // try to save old data
547  if (tai_memcpy_to_kernel(pid, saved, dest, size) < 0) {
548  LOG("Invalid address for memcpy");
549  sceKernelDeleteUid(ret);
550  sceKernelMemPoolFree(g_patch_pool, saved);
551  return TAI_ERROR_INVALID_ARGS;
552  }
553 
554  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
555  patch->type = INJECTION;
556  patch->uid = ret;
557  patch->pid = pid;
558  patch->addr = (uintptr_t)dest;
559  patch->size = size;
560  patch->next = NULL;
561  patch->data.inject.saved = saved;
562  patch->data.inject.size = size;
563  patch->data.inject.patch = patch;
564  if (proc_map_try_insert(g_map, patch, &tmp) < 1) {
565  ret = TAI_ERROR_PATCH_EXISTS;
566  } else {
567  ret = tai_force_memcpy(pid, dest, src, size);
568  }
569 
570  if (ret < 0) {
571  sceKernelDeleteUid(patch->uid);
572  sceKernelMemPoolFree(g_patch_pool, saved);
573  } else {
574  ret = patch->uid;
575  }
576 
577  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
578 
579  return ret;
580 }

644 int tai_try_cleanup_process(SceUID pid) {
645  tai_patch_t *patch, *next;
646  tai_hook_t *hook, *nexthook;
647  LOG("Calling patches cleanup for pid %x", pid);
648  sceKernelLockMutexForKernel(g_hooks_lock, 1, NULL);
649  if (proc_map_remove_all_pid(g_map, pid, &patch) > 0) {
650  while (patch != NULL) {
651  next = patch->next;
652  if (patch->type == HOOKS) {
653  hook = patch->data.hooks.head;
654  while (hook != NULL) {
655  nexthook = hook->next;
656  slab_free(patch->slab, hook);
657  hook = nexthook;
658  }
659  }
660  sceKernelDeleteUid(patch->uid);
661  patch = next;
662  }
663  }
664  sceKernelUnlockMutexForKernel(g_hooks_lock, 1);
665  return 0;
