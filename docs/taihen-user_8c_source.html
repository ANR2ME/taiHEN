41 SceUID taiHookFunctionExportForUser(tai_hook_ref_t *p_hook, tai_hook_args_t *args) {
42  tai_hook_args_t kargs;
43  uint32_t func_nid;
44  const void *hook_func;
45  uint32_t state;
46  char k_module[MAX_NAME_LEN];
47  tai_hook_ref_t k_ref;
48  SceUID kid, ret;
49  SceUID pid;
50 
51  ENTER_SYSCALL(state);
52  kargs.size = 0;
53  sceKernelMemcpyUserToKernel(&kargs, (uintptr_t)args, sizeof(kargs));
54  if (kargs.size == sizeof(kargs)) {
55  pid = sceKernelGetProcessId();
56  if (sceKernelStrncpyUserToKernel(k_module, (uintptr_t)kargs.module, MAX_NAME_LEN) < MAX_NAME_LEN) {
57  kid = taiHookFunctionExportForKernel(pid, &k_ref, k_module, kargs.library_nid, kargs.func_nid, kargs.hook_func);
58  if (kid >= 0) {
59  sceKernelMemcpyKernelToUser((uintptr_t)p_hook, &k_ref, sizeof(*p_hook));
60  ret = sceKernelCreateUserUid(pid, kid);
61  LOG("kernel uid: %x, user uid: %x", kid, ret);
62  } else {
63  ret = kid;
64  }
65  } else {
66  ret = TAI_ERROR_USER_MEMORY;
67  }
68  } else {
69  LOG("invalid args size: %x", kargs.size);
70  ret = TAI_ERROR_USER_MEMORY;
71  }
72  EXIT_SYSCALL(state);
73  return ret;
74 }
