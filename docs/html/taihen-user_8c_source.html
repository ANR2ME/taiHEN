45 SceUID taiHookFunctionExport(tai_hook_ref_t *p_hook, const char *module, uint32_t library_nid, uint32_t func_nid, const void *hook_func) {
46  uint32_t state;
47  char k_module[MAX_NAME_LEN];
48  tai_hook_ref_t k_ref;
49  SceUID kid, ret;
50  SceUID pid;
51 
52  ENTER_SYSCALL(state);
53  pid = sceKernelGetProcessId();
54  if (sceKernelStrncpyUserToKernel(k_module, (uintptr_t)module, MAX_NAME_LEN) < MAX_NAME_LEN) {
55  kid = taiHookFunctionExportForKernel(pid, &k_ref, k_module, library_nid, func_nid, hook_func);
56  if (kid >= 0) {
57  sceKernelMemcpyKernelToUser((uintptr_t)p_hook, &k_ref, sizeof(*p_hook));
58  ret = sceKernelCreateUserUid(pid, kid);
59  LOG("1kernel uid: %x, user uid: %x", kid, ret);
60  ret = sceKernelCreateUserUid(pid, kid);
61  LOG("2kernel uid: %x, user uid: %x", kid, ret);
62  ret = sceKernelCreateUserUid(pid, kid);
63  LOG("3kernel uid: %x, user uid: %x", kid, ret);
64  } else {
65  ret = kid;
66  }
67  } else {
68  ret = TAI_ERROR_USER_MEMORY;
69  }
70  EXIT_SYSCALL(state);
71  return ret;
72 }
