144 static inline uint32_t next_pow_2(uint32_t v) {
145  v--;
146  v |= v >> 1;
147  v |= v >> 2;
148  v |= v >> 4;

156 void slab_init(struct slab_chain *const sch, const size_t itemsize, SceUID pid)
157 {
158  assert(sch != NULL);
159  assert(itemsize >= 1 && itemsize <= SIZE_MAX);
160  assert(POWEROF2(slab_pagesize));
161 
162  sch->itemsize = itemsize;
163  sch->pid = pid;
164 
165  const size_t data_offset = offsetof(struct slab_header, data);
166  const size_t least_slabsize = data_offset + 64 * sch->itemsize;
167  sch->slabsize = (size_t) next_pow_2(least_slabsize);
168  sch->itemcount = 64;
169 
170  if (sch->slabsize - least_slabsize != 0) {
171  const size_t shrinked_slabsize = sch->slabsize >> 1;
172 
173  if (data_offset < shrinked_slabsize &&
174  shrinked_slabsize - data_offset >= 2 * sch->itemsize) {
175 
176  sch->slabsize = shrinked_slabsize;
177  sch->itemcount = (shrinked_slabsize - data_offset) / sch->itemsize;
178  }
179  }
180 
181  sch->pages_per_alloc = sch->slabsize > slab_pagesize ?
182  sch->slabsize : slab_pagesize;
183 
184  sch->empty_slotmask = ~SLOTS_ALL_ZERO >> (64 - sch->itemcount);
185  sch->initial_slotmask = sch->empty_slotmask ^ SLOTS_FIRST;
186  sch->alignment_mask = ~(sch->slabsize - 1);
187  sch->partial = sch->empty = sch->full = NULL;
188 
189  assert(slab_is_valid(sch));
190 }

192 void *slab_alloc(struct slab_chain *const sch, uintptr_t *exe_addr)
193 {
194  assert(sch != NULL);
195  assert(slab_is_valid(sch));
196 
197  if (LIKELY(sch->partial != NULL)) {
198  /* found a partial slab, locate the first free slot */
199  register const size_t slot = FIRST_FREE_SLOT(sch->partial->slots);
200  sch->partial->slots ^= SLOTS_FIRST << slot;
201 
202  if (UNLIKELY(sch->partial->slots == SLOTS_ALL_ZERO)) {
203  /* slab has become full, change state from partial to full */
204  struct slab_header *const tmp = sch->partial;
205 
206  /* skip first slab from partial list */
207  if (LIKELY((sch->partial = sch->partial->next) != NULL))
208  sch->partial->prev = NULL;
209 
210  if (LIKELY((tmp->next = sch->full) != NULL))
211  sch->full->prev = tmp;
212 
213  sch->full = tmp;
214  *exe_addr = sch->full->exe_data + slot * sch->itemsize;
215  return sch->full->data + slot * sch->itemsize;
216  } else {
217  *exe_addr = sch->partial->exe_data + slot * sch->itemsize;
218  return sch->partial->data + slot * sch->itemsize;
219  }
220  } else if (LIKELY((sch->partial = sch->empty) != NULL)) {
221  /* found an empty slab, change state from empty to partial */
222  if (LIKELY((sch->empty = sch->empty->next) != NULL))
223  sch->empty->prev = NULL;
224 
225  sch->partial->next = NULL;
226 
227  /* slab is located either at the beginning of page, or beyond */
228  UNLIKELY(sch->partial->refcount != 0) ?
229  sch->partial->refcount++ : sch->partial->page->refcount++;
230 
231  sch->partial->slots = sch->initial_slotmask;
232  *exe_addr = sch->partial->exe_data;
233  return sch->partial->data;
234  } else {
235  /* no empty or partial slabs available, create a new one */
236  SceUID write_res, exe_res;
237  uintptr_t exe_data;
238  if ((write_res = sce_exe_alloc(sch->pid, (void **)&sch->partial, &exe_data,
239  &exe_res, sch->slabsize, sch->pages_per_alloc)) < 0) {
240  *exe_addr = 0;
241  return sch->partial = NULL;
242  }
243  sch->partial->write_res = write_res;
244  sch->partial->exe_res = exe_res;
245  sch->partial->exe_data = exe_data + offsetof(struct slab_header, data);
246  exe_data += sch->slabsize;
247 
248  struct slab_header *prev = NULL;
249 
250  const char *const page_end =
251  (char *) sch->partial + sch->pages_per_alloc;
252 
253  union {
254  const char *c;
255  struct slab_header *const s;
256  } curr = {
257  .c = (const char *) sch->partial + sch->slabsize
258  };
259 
260  __builtin_prefetch(sch->partial, 1);
261 
262  sch->partial->prev = sch->partial->next = NULL;
263  sch->partial->refcount = 1;
264  sch->partial->slots = sch->initial_slotmask;
265 
266  if (LIKELY(curr.c != page_end)) {
267  curr.s->prev = NULL;
268  curr.s->refcount = 0;
269  curr.s->page = sch->partial;
270  curr.s->write_res = write_res;
271  curr.s->exe_res = exe_res;
272  curr.s->exe_data = exe_data;
273  exe_data += sch->slabsize;
274  curr.s->slots = sch->empty_slotmask;
275  sch->empty = prev = curr.s;
276 
277  while (LIKELY((curr.c += sch->slabsize) != page_end)) {
278  prev->next = curr.s;
279  curr.s->prev = prev;
280  curr.s->refcount = 0;
281  curr.s->page = sch->partial;
282  curr.s->write_res = write_res;
283  curr.s->exe_res = exe_res;
284  curr.s->exe_data = exe_data;
285  exe_data += sch->slabsize;
286  curr.s->slots = sch->empty_slotmask;
287  prev = curr.s;
288  }
289 
290  prev->next = NULL;
291  }
292 
293  *exe_addr = sch->partial->exe_data;
294  return sch->partial->data;
295  }
296 
297  /* unreachable */
298 }

300 void slab_free(struct slab_chain *const sch, const void *const addr)
301 {
302  assert(sch != NULL);
303  assert(slab_is_valid(sch));
304  assert(addr != NULL);
305 
306  struct slab_header *const slab = (void *)
307  ((uintptr_t) addr & sch->alignment_mask);
308 
309  register const int slot = ((char *) addr - (char *) slab -
310  offsetof(struct slab_header, data)) / sch->itemsize;
311 
312  if (UNLIKELY(slab->slots == SLOTS_ALL_ZERO)) {
313  /* target slab is full, change state to partial */
314  slab->slots = SLOTS_FIRST << slot;
315 
316  if (LIKELY(slab != sch->full)) {
317  if (LIKELY((slab->prev->next = slab->next) != NULL))
318  slab->next->prev = slab->prev;
319 
320  slab->prev = NULL;
321  } else if (LIKELY((sch->full = sch->full->next) != NULL)) {
322  sch->full->prev = NULL;
323  }
324 
325  slab->next = sch->partial;
326 
327  if (LIKELY(sch->partial != NULL))
328  sch->partial->prev = slab;
329 
330  sch->partial = slab;
331  } else if (UNLIKELY(ONE_USED_SLOT(slab->slots, sch->empty_slotmask))) {
332  /* target slab is partial and has only one filled slot */
333  if (UNLIKELY(slab->refcount == 1 || (slab->refcount == 0 &&
334  slab->page->refcount == 1))) {
335 
336  /* unmap the whole page if this slab is the only partial one */
337  if (LIKELY(slab != sch->partial)) {
338  if (LIKELY((slab->prev->next = slab->next) != NULL))
339  slab->next->prev = slab->prev;
340  } else if (LIKELY((sch->partial = sch->partial->next) != NULL)) {
341  sch->partial->prev = NULL;
342  }
343 
344  void *const page = UNLIKELY(slab->refcount != 0) ? slab : slab->page;
345  const char *const page_end = (char *) page + sch->pages_per_alloc;
346  char found_head = 0;
347 
348  union {
349  const char *c;
350  const struct slab_header *const s;
351  } s;
352 
353  for (s.c = page; s.c != page_end; s.c += sch->slabsize) {
354  if (UNLIKELY(s.s == sch->empty))
355  found_head = 1;
356  else if (UNLIKELY(s.s == slab))
357  continue;
358  else if (LIKELY((s.s->prev->next = s.s->next) != NULL))
359  s.s->next->prev = s.s->prev;
360  }
361 
362  if (UNLIKELY(found_head && (sch->empty = sch->empty->next) != NULL))
363  sch->empty->prev = NULL;
364 
365  sce_exe_free(slab->write_res, slab->exe_res);
366  } else {
367  slab->slots = sch->empty_slotmask;
368 
369  if (LIKELY(slab != sch->partial)) {
370  if (LIKELY((slab->prev->next = slab->next) != NULL))
371  slab->next->prev = slab->prev;
372 
373  slab->prev = NULL;
374  } else if (LIKELY((sch->partial = sch->partial->next) != NULL)) {
375  sch->partial->prev = NULL;
376  }
377 
378  slab->next = sch->empty;
379 
380  if (LIKELY(sch->empty != NULL))
381  sch->empty->prev = slab;
382 
383  sch->empty = slab;
384 
385  UNLIKELY(slab->refcount != 0) ?
386  slab->refcount-- : slab->page->refcount--;
387  }
388  } else {
389  /* target slab is partial, no need to change state */
390  slab->slots |= SLOTS_FIRST << slot;
391  }
392 }

407 void slab_traverse(const struct slab_chain *const sch, void (*fn)(const void *))
408 {
409  assert(sch != NULL);
410  assert(fn != NULL);
411  assert(slab_is_valid(sch));
412 
413  const struct slab_header *slab;
414  const char *item, *end;
415  const size_t data_offset = offsetof(struct slab_header, data);
416 
417  for (slab = sch->partial; slab; slab = slab->next) {
418  item = (const char *) slab + data_offset;
419  end = item + sch->itemcount * sch->itemsize;
420  uint64_t mask = SLOTS_FIRST;
421 
422  do {
423  if (!(slab->slots & mask))
424  fn(item);
425 
426  mask <<= 1;
427  } while ((item += sch->itemsize) != end);
428  }
429 
430  for (slab = sch->full; slab; slab = slab->next) {
431  item = (const char *) slab + data_offset;
432  end = item + sch->itemcount * sch->itemsize;
433 
434  do fn(item);
435  while ((item += sch->itemsize) != end);
436  }

440 {

443 
444  struct slab_header *const heads[] = {sch->partial, sch->empty, sch->full};
445  struct slab_header *pages_head = NULL, *pages_tail;
446 
447  for (size_t i = 0; i < 3; ++i) {
448  struct slab_header *slab = heads[i];
449 
450  while (slab != NULL) {
451  if (slab->refcount != 0) {
452  struct slab_header *const page = slab;
453  slab = slab->next;
454 
455  if (UNLIKELY(pages_head == NULL))
456  pages_head = page;
457  else
458  pages_tail->next = page;
459 
460  pages_tail = page;
461  } else {
462  slab = slab->next;
463  }
464  }
465  }
466

468  pages_tail->next = NULL;
