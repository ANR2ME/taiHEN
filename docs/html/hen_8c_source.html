127 static int parse_headers_patched(int ctx, const void *headers, size_t len, void *args) {
128  self_header_t *self;
129  Elf32_Ehdr *elf;
130  int ret;
131  int num_segs;
132 
133  memset(&g_seg_info, 0, sizeof(g_seg_info));
134  if (len >= sizeof(self_header_t) && len >= sizeof(Elf32_Ehdr)) {
135  self = (self_header_t *)headers;
136  if (self->elf_offset <= len - sizeof(Elf32_Ehdr)) {
137  elf = (Elf32_Ehdr *)(headers + self->elf_offset);
138  num_segs = elf->e_phnum;
139  if (num_segs <= MAX_SEGMENTS &&
140  self->section_info_offset < self->section_info_offset + num_segs * sizeof(self_section_info_t) &&
141  self->section_info_offset + num_segs * sizeof(self_section_info_t) < len
142  ) {
143  memcpy(&g_seg_info, headers + self->section_info_offset, num_segs * sizeof(self_section_info_t));
144  }
145  }
146  }
147  ret = TAI_CONTINUE(int, g_parse_headers_hook, ctx, headers, len, args);
148  if (ctx == 1) { // as of 3.60, only one decrypt context exists
149  if (ret == 0x800f0624 || ret == 0x800f0616 || ret == 0x800f0024 || ret == 0x800f0b3a) {
150  g_is_homebrew = 1;
151  // we only do this patch if another hook hasn't modified it already
152  if (*(uint32_t *)(args + OFFSET_PATCH_ARG) == 0) {
153  *(uint32_t *)(args + OFFSET_PATCH_ARG) = 0x20;
154  }
155  ret = 0;
156  } else {
157  g_is_homebrew = 0;
158  }
159  LOG("parse ret %x, decrypt is homebrew? %d", ret, g_is_homebrew);
160  }
161  return ret;
162 }
